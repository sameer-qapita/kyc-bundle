 <script defer>
      fetch("https://open-marketplacedev.qapita.dev/api/v1/d/listings")
        .then(response => response.json())
        .then(publicData => {
          const carouselElement = document.querySelector('.custom-slide-track');
    
          const carouselContent = publicData.data.map(company => `
            <div class="slide">
             <a href="/outreach/d/listings/${company.id}/view">

              <p>${company.name}</p>
              </a>
              
            </div>
          `).join('');
    
          carouselElement.innerHTML = carouselContent;
    
publicData.data.forEach((company, index) => {
    const organizationId = company.organizationId;
    const slideElement = carouselElement.querySelectorAll('.slide')[index].querySelector('img');

    fetch(`https://dbx5fu8doywo9.cloudfront.net/api/v1/d/organizations/${organizationId}/logo`)
     .then(response => {
            if (response.status === 200) {
              return response.text();
            } else {
              return 'https://uploads-ssl.webflow.com/633e5454c8dec0fa090556c5/651f04affb63f9233626f84c_no-logo.png';
            }
        })
         .then(srcPath => {
              const img = document.createElement('img');
              img.src = srcPath;
						  const slideElement = document.querySelectorAll('.slide')[index].querySelector('a');
              slideElement.appendChild(img); 
          })
          .catch(error => 
            console.log('failed to fetch logo',error));
            });

        })
        .catch(error => console.error('Error fetching data:', error));
    </script>
